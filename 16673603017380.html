Debian 软件环境配置

2022/11/02 posted in  Debian

安装Docker

卸载原先的，不管有没有

$ sudo apt-get remove docker docker-engine docker.io containerd runc

更新 apt 包索引

$ sudo apt-get update
 
$ sudo apt-get install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release

安装GPG证书

curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

装一个docker源

echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/debian \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sour2022年11月3日ces.list.d/docker.list > /dev/null

安装最新版本的 Docker Engine-Community 和 containerd

$ sudo apt install docker-ce docker-ce-cli containerd.io

校对版本
查看一下库里有哪些版本

$ apt-cache   madison   docker-ce

再看一下自己装的啥版本

$ docker --version

$ docker info

查看Docker运行状态

$ systemctl  status  docker

更改Docker镜像源

卸载 docker

删除安装包：

sudo apt-get purge docker-ce

删除镜像、容器、配置文件等内容

sudo rm -rf /var/lib/docker

安装docker-compose

sudo pip install docker-compose

安装git

sudo apt install git

修改Hosts文件

sudo vim /etc/hosts

# 按ip 域名格式加上github
140.82.114.3 github.com
146.75.77.194 github.global.ssl.fastly.net

github对应的IP可以通过Best IP来查询

刷新DNS缓存

sudo apt install nscd # 如果没有nscd需要先下载
sudo /etc/init.d/nscd restart

安装zsh

# 依赖的包也会安装
sudo apt install zsh

chsh -s /bin/zsh #将zsh设置成默认shell（不设置的话启动zsh只有直接zsh命令即可）

root 重启生效

安装oh-my-zsh

sh -c "$(curl -fsSL https://gitee.com/shmhlsy/oh-my-zsh-install.sh/raw/master/install.sh)" #国内镜像源

安装zplug

sudo apt install zplug

改动~/.zshrc

# If you come from bash you might have to change your $PATH.
export PATH=$HOME/bin:/usr/local/bin:$PATH

export ZSH="$HOME/.oh-my-zsh"
export ZPLUG_HOME=/usr/share/zplug

# 启动 zplug
source $ZPLUG_HOME/init.zsh
source $ZSH/oh-my-zsh.sh

# 安装theme
zplug "themes/amuse", as:theme, from:oh-my-zsh, use:amuse.zsh-theme

# 安装plugin
zplug "plugins/git",   from:oh-my-zsh
zplug "plugins/vi-mode", from:oh-my-zsh
zplug "plugins/extract", from:oh-my-zsh
zplug "zsh-users/zsh-completions"
zplug "zsh-users/zsh-history-substring-search"
zplug "zsh-users/zsh-autosuggestions"
zplug "zdharma/fast-syntax-highlighting"

# 如果遇到沒有安装过的Plugin，就会提醒安装、启动
if ! zplug check --verbose; then
    printf "Install? [y/N]: "
    if read -q; then
        echo; zplug install
    else
        echo
    fi
fi

zplug load

配置生效，自动下载插件

sudo chmod 777 /usr/share/zplug # 配置权限
source ~/.zshrc
